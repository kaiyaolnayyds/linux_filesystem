cmake_minimum_required(VERSION 3.10)

# 设置项目名称
project(SimDiskFS)

# 设置 C++ 标准
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# 添加源文件
set(SOURCES
    main.cpp
    DiskManager.cpp
    DiskManager.h
    SuperBlock.cpp
    SuperBlock.h
    INode.cpp
    INode.h
    Directory.cpp
    Directory.h
    UserManager.cpp
    UserManager.h
    CommandHandler.cpp
    CommandHandler.h
)

# 生成可执行文件
add_executable(simdisk ${SOURCES})

# 添加编译选项（可选）
target_compile_options(simdisk PRIVATE -Wall -Wextra -Wpedantic)
